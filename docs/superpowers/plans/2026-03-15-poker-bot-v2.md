> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a new poker bot from scratch using precomputed exact equity, adaptive opponent modeling, GTO-anchored bluffing, and counter-intelligence defenses to maximize tournament win rate.

**Architecture:** Six modules (`card_utils.py`, `equity.py`, `opponent.py`, `strategy.py`, `match_manager.py`, `player.py`) plus an offline table generator (`generate_tables.py`) and three precomputed `.npy` tables. The bot replaces MC sampling with O(1) table lookups, uses sigmoid probabilistic thresholds everywhere, and adapts to opponent tendencies within each match.

**Tech Stack:** Python 3.12+, NumPy (array ops + table lookups), treys (hand evaluation via WrappedEval), FastAPI/uvicorn (inherited from Agent base class)

```python
# submission/card_utils.py
"""Card encoding, combinatorial indexing, and hand classification for 27-card poker."""

NUM_RANKS = 9   # 2,3,4,5,6,7,8,9,A
NUM_SUITS = 3   # d,h,s
NUM_CARDS = 27

# ── Card encoding ────────────────────────────────────────────────────────────
# card_int = suit_index * 9 + rank_index
# rank_index: 0=2, 1=3, ..., 7=9, 8=Ace
# suit_index: 0=diamonds, 1=hearts, 2=spades

def rank(card: int) -> int:
    return card % NUM_RANKS

def suit(card: int) -> int:
    return card // NUM_RANKS

def is_suited(c1: int, c2: int) -> bool:
    return suit(c1) == suit(c2)

def is_paired(c1: int, c2: int) -> bool:
    return rank(c1) == rank(c2)

def is_connected(c1: int, c2: int) -> bool:
    r1, r2 = rank(c1), rank(c2)
    diff = abs(r1 - r2)
    if diff <= 2:
        return True
    # Ace (rank 8) wraps to low: A-2 (diff=8), A-3 (diff=7) are connected
    # because A-2-3-4-5 is a valid straight in standard poker
    # Also 9-A (diff=1) already handled above
    # In this 27-card deck, A can act as 10 for 6-7-8-9-A straights
    # So A connects with 7,8,9 (high) and 2,3 (low)
    if r1 == 8 or r2 == 8:  # one is Ace
        other = r2 if r1 == 8 else r1
        return other <= 1 or other >= 6  # 2,3 (low wrap) or 7,8,9 (high)
    return False

def is_high(card: int) -> bool:
    return rank(card) >= 6  # 8, 9, or Ace

# ── Combinatorial indexing ───────────────────────────────────────────────────
# Combinatorial number system: maps sorted k-card tuples to unique indices

# Precompute C(n, k) for n=0..27, k=0..7
COMB = [[0] * 8 for _ in range(28)]
for _n in range(28):
    COMB[_n][0] = 1
    for _k in range(1, min(_n + 1, 8)):
        COMB[_n][_k] = COMB[_n - 1][_k - 1] + COMB[_n - 1][_k]

def combo_index(cards: tuple, k: int) -> int:
    """Map sorted k-card combo to unique index in [0, C(27,k))."""
    s = sorted(cards)
    return sum(COMB[s[i]][i + 1] for i in range(k))

def combo_index_2(cards: tuple) -> int:
    return combo_index(cards, 2)

def combo_index_5(cards: tuple) -> int:
    return combo_index(cards, 5)

def combo_index_7(cards: tuple) -> int:
    return combo_index(cards, 7)

# ── Hand classification ──────────────────────────────────────────────────────
# WrappedEval score ranges (lower = stronger):

def classify_hand(score: int) -> str:
    if score <= 10:    return "straight_flush"
    if score <= 322:   return "full_house"
    if score <= 1599:  return "flush"
    if score <= 1609:  return "straight"
    if score <= 2467:  return "trips"
    if score <= 3325:  return "two_pair"
    if score <= 6185:  return "pair"
    return "high_card"
```

```python
# submission/generate_tables.py
"""
Offline table generator for poker bot v2.
Generates precomputed .npy lookup tables.

Usage:
    python -m submission.generate_tables --table hand_ranks
    python -m submission.generate_tables --table hand_strength_2
    python -m submission.generate_tables --table all

NOT shipped in tournament submission — only the .npy files are shipped.
"""
import os
import sys
import time
import argparse
import numpy as np
from itertools import combinations

# Add parent dir to path so we can import gym_env
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from gym_env import PokerEnv
from submission.card_utils import combo_index_7, combo_index_2, combo_index_5, COMB, NUM_CARDS

TABLE_DIR = os.path.join(os.path.dirname(os.path.abspath(__file__)), 'tables')


def generate_hand_ranks():
    """Generate hand_ranks.npy: maps every 7-card combo to WrappedEval score."""
    os.makedirs(TABLE_DIR, exist_ok=True)
    evaluator = PokerEnv().evaluator
    int_to_card = PokerEnv.int_to_card

    total = COMB[NUM_CARDS][7]  # 888030
    table = np.zeros(total, dtype=np.uint16)

    print(f"Generating hand_ranks.npy ({total} entries)...")
    t0 = time.time()

    for i, combo in enumerate(combinations(range(NUM_CARDS), 7)):
        # Any 2+5 split gives same score (partition-independent)
        hand = [int_to_card(combo[0]), int_to_card(combo[1])]
        board = [int_to_card(c) for c in combo[2:]]
        score = evaluator.evaluate(hand, board)
        idx = combo_index_7(combo)
        table[idx] = score

        if (i + 1) % 100000 == 0:
            elapsed = time.time() - t0
            rate = (i + 1) / elapsed
            eta = (total - i - 1) / rate
            print(f"  {i+1}/{total} ({100*(i+1)/total:.1f}%) — {rate:.0f} combos/sec, ETA {eta:.0f}s")

    elapsed = time.time() - t0
    path = os.path.join(TABLE_DIR, 'hand_ranks.npy')
    np.save(path, table)
    print(f"Saved {path} ({table.nbytes} bytes) in {elapsed:.1f}s")

    # Validate boundary indices
    assert combo_index_7((0,1,2,3,4,5,6)) == 0
    assert combo_index_7((20,21,22,23,24,25,26)) == total - 1
    assert 1 <= table[0] <= 7462
    assert 1 <= table[total - 1] <= 7462
    print("Validation passed.")
    return table


def generate_hand_strength_2(hand_ranks=None):
    """Generate hand_strength_2.npy: average equity of each 2-card hand."""
    os.makedirs(TABLE_DIR, exist_ok=True)

    if hand_ranks is None:
        hr_path = os.path.join(TABLE_DIR, 'hand_ranks.npy')
        if not os.path.exists(hr_path):
            print("hand_ranks.npy not found — generating it first...")
            hand_ranks = generate_hand_ranks()
        else:
            hand_ranks = np.load(hr_path)

    total_2 = COMB[NUM_CARDS][2]  # 351
    table = np.zeros(total_2, dtype=np.float32)

    print(f"Generating hand_strength_2.npy ({total_2} entries)...")
    t0 = time.time()

    for i, my_hand in enumerate(combinations(range(NUM_CARDS), 2)):
        remaining = [c for c in range(NUM_CARDS) if c not in my_hand]
        wins = 0
        ties = 0
        total_matchups = 0

        for board in combinations(remaining, 5):
            my_idx = combo_index_7(tuple(sorted(my_hand + board)))
            my_score = hand_ranks[my_idx]

            opp_remaining = [c for c in remaining if c not in board]
            for opp_hand in combinations(opp_remaining, 2):
                opp_idx = combo_index_7(tuple(sorted(opp_hand + board)))
                opp_score = hand_ranks[opp_idx]
                if my_score < opp_score:
                    wins += 1
                elif my_score == opp_score:
                    ties += 1
                total_matchups += 1

        equity = (wins + 0.5 * ties) / total_matchups if total_matchups > 0 else 0.5
        idx = combo_index_2(my_hand)
        table[idx] = equity

        if (i + 1) % 50 == 0:
            elapsed = time.time() - t0
            print(f"  {i+1}/{total_2} ({100*(i+1)/total_2:.1f}%) — {elapsed:.1f}s elapsed")

    elapsed = time.time() - t0
    path = os.path.join(TABLE_DIR, 'hand_strength_2.npy')
    np.save(path, table)
    print(f"Saved {path} ({table.nbytes} bytes) in {elapsed:.1f}s")

    # Sanity: equity values in [0, 1]
    assert table.min() >= 0.0 and table.max() <= 1.0
    print(f"Equity range: [{table.min():.4f}, {table.max():.4f}], mean={table.mean():.4f}")
    return table


if __name__ == '__main__':
    parser = argparse.ArgumentParser(description='Generate precomputed tables')
    parser.add_argument('--table', choices=['hand_ranks', 'hand_strength_2', 'all'],
                        default='all', help='Which table to generate')
    args = parser.parse_args()

    if args.table in ('hand_ranks', 'all'):
        hr = generate_hand_ranks()
    else:
        hr = None

    if args.table in ('hand_strength_2', 'all'):
        generate_hand_strength_2(hr)
```

```python
# submission/equity.py
"""
Exact equity engine using precomputed hand rank tables.
Zero Monte Carlo sampling — all equity is computed via exhaustive enumeration
with O(1) table lookups.
"""
import os
import math
import random
import numpy as np
from itertools import combinations
from submission.card_utils import (
    combo_index_2, combo_index_5, combo_index_7,
    NUM_CARDS, COMB, rank, suit
)

# ── Load precomputed tables ──────────────────────────────────────────────────
_TABLE_DIR = os.path.join(os.path.dirname(os.path.abspath(__file__)), 'tables')

def _load_table(name, dtype=None):
    path = os.path.join(_TABLE_DIR, name)
    if os.path.exists(path):
        return np.load(path)
    return None

HAND_RANKS = _load_table('hand_ranks.npy')        # (888030,) uint16
HAND_STRENGTH_2 = _load_table('hand_strength_2.npy')  # (351,) float32
HAND_POTENTIAL_5 = _load_table('hand_potential_5.npy')  # (80730,) float32 — may not exist yet


# ── Core functions ───────────────────────────────────────────────────────────

def hand_rank(my_2: tuple, board_5: tuple) -> int:
    """Exact hand score from precomputed table. O(1)."""
    idx = combo_index_7(tuple(sorted(my_2 + board_5)))
    return int(HAND_RANKS[idx])


def hand_strength_2_lookup(hand: tuple) -> float:
    """Average equity of a 2-card hand across all boards/opponents."""
    if HAND_STRENGTH_2 is None:
        return 0.5
    return float(HAND_STRENGTH_2[combo_index_2(tuple(sorted(hand)))])


def preflop_strength(hole_5: tuple) -> float:
    """Pre-flop potential of a 5-card starting hand. O(1) lookup."""
    if HAND_POTENTIAL_5 is not None:
        return float(HAND_POTENTIAL_5[combo_index_5(tuple(sorted(hole_5)))])
    # Fallback: average hand_strength_2 of all 10 possible keeps
    if HAND_STRENGTH_2 is None:
        return 0.5
    best = 0.0
    for i, j in combinations(range(5), 2):
        keep = (hole_5[i], hole_5[j])
        eq = hand_strength_2_lookup(keep)
        if eq > best:
            best = eq
    return best


def exact_equity(my_2, board, dead_cards, opp_discards=None,
                 my_discards=None, opp_weights_fn=None) -> float:
    """
    Exact equity via enumeration of opponent hands.

    When opp_discards is provided, applies the rationality filter:
    only considers opponent hands that would be the rational keep
    from their original 5 cards given the actual flop.

    Args:
        my_2: our 2 kept cards (tuple of 2 ints)
        board: known community cards (tuple of 3-5 ints)
        dead_cards: all known dead cards (tuple/list of ints)
        opp_discards: opponent's 3 revealed discards (for rationality filter)
        my_discards: our 3 discards (opponent saw these — dead for their eval)
        opp_weights_fn: function(opp_2_card_tuple) → float weight

    Returns: win probability [0.0, 1.0]
    """
    all_dead = set(my_2) | set(board) | set(dead_cards or ())
    remaining = [c for c in range(NUM_CARDS) if c not in all_dead]

    if len(board) == 5:
        # River: just enumerate opponent 2-card hands
        return _equity_river(my_2, board, remaining, opp_discards,
                             my_discards, opp_weights_fn)
    elif len(board) == 4:
        # Turn: enumerate 1 more board card + opponent hands
        return _equity_partial_board(my_2, board, remaining, 1,
                                     opp_discards, my_discards, opp_weights_fn)
    elif len(board) == 3:
        # Flop: enumerate 2 more board cards + opponent hands
        return _equity_partial_board(my_2, board, remaining, 2,
                                     opp_discards, my_discards, opp_weights_fn)
    else:
        return 0.5  # pre-flop — use preflop_strength() instead


def _equity_river(my_2, board_5, remaining, opp_discards, my_discards, opp_weights_fn):
    """Exact equity when all 5 board cards are known."""
    my_score = hand_rank(my_2, tuple(board_5))
    wins = 0.0
    total_weight = 0.0

    if opp_discards is not None:
        opp_hands = list(_rational_opponent_hands(remaining, opp_discards,
                                                   tuple(board_5[:3]), my_discards))
    else:
        opp_hands = list(combinations(remaining, 2))

    for opp_hand in opp_hands:
        w = opp_weights_fn(opp_hand) if opp_weights_fn else 1.0
        if w <= 0:
            continue
        opp_score = hand_rank(opp_hand, tuple(board_5))
        if my_score < opp_score:
            wins += w
        elif my_score == opp_score:
            wins += 0.5 * w
        total_weight += w

    return wins / total_weight if total_weight > 0 else 0.5


def _equity_partial_board(my_2, board_partial, remaining, cards_needed,
                          opp_discards, my_discards, opp_weights_fn):
    """Exact equity when board is incomplete (flop or turn)."""
    wins = 0.0
    total_weight = 0.0

    # Pre-filter rational opponent hands using the flop (first 3 board cards)
    flop = tuple(board_partial[:3])
    if opp_discards is not None:
        rational_hands = list(_rational_opponent_hands(remaining, opp_discards,
                                                        flop, my_discards))
    else:
        rational_hands = None

    for board_ext in combinations(remaining, cards_needed):
        full_board = tuple(board_partial) + board_ext
        my_score = hand_rank(my_2, full_board)

        opp_remaining = [c for c in remaining if c not in board_ext]
        opp_hands = rational_hands if rational_hands is not None else list(combinations(opp_remaining, 2))

        # Filter out hands that overlap with board extension
        board_ext_set = set(board_ext)
        for opp_hand in opp_hands:
            if opp_hand[0] in board_ext_set or opp_hand[1] in board_ext_set:
                continue
            w = opp_weights_fn(opp_hand) if opp_weights_fn else 1.0
            if w <= 0:
                continue
            opp_score = hand_rank(opp_hand, full_board)
            if my_score < opp_score:
                wins += w
            elif my_score == opp_score:
                wins += 0.5 * w
            total_weight += w

    return wins / total_weight if total_weight > 0 else 0.5


# ── Rationality filter ───────────────────────────────────────────────────────

def _rational_opponent_hands(remaining, opp_discards, flop, my_discards=None):
    """
    Yield only opponent hands that would be the rational keep
    from their original 5 cards given the actual flop.

    The opponent saw our discards before choosing — our discards
    are dead cards in their evaluation.
    """
    opp_disc_list = list(opp_discards)
    extra_dead = list(my_discards) if my_discards else []

    for hand in combinations(remaining, 2):
        original_5 = list(hand) + opp_disc_list
        if len(original_5) != 5:
            continue

        best_eq = -1.0
        best_keep_set = None

        for i, j in combinations(range(5), 2):
            keep = (original_5[i], original_5[j])
            discard = [original_5[k] for k in range(5) if k not in (i, j)]
            # Evaluate this keep against the actual flop
            # Dead cards = the discard + extra dead (our discards)
            dead = set(discard) | set(extra_dead) | set(flop)
            opp_remaining = [c for c in range(NUM_CARDS) if c not in dead and c not in keep]

            # Quick equity: enumerate opponent of opponent on the flop
            eq = _quick_flop_equity(keep, flop, opp_remaining)
            if eq > best_eq:
                best_eq = eq
                best_keep_set = set(keep)

        if best_keep_set == set(hand):
            yield hand


def _quick_flop_equity(keep_2, flop_3, remaining):
    """
    Equity estimate for rationality filter keep evaluation.

    KNOWN LIMITATION (v2 launch): Uses hand_strength_2 (pre-flop averages)
    as a fast proxy. The spec requires flop-aware exact_equity here, but
    that creates a ~1-2 second Python-loop bottleneck per hand.

    This means the filter will occasionally mis-rank keeps (e.g., flush
    draws on flush-heavy boards may be undervalued vs pairs). This is
    the SAME approximation error the spec warns about, accepted here as
    a launch tradeoff.

    Task 13 (vectorize equity engine) will fix this: once the inner loop
    is vectorized with NumPy, we can afford flop-aware evaluation here.
    Until then, this is a ~5% inaccuracy in filter decisions, which is
    far better than no rationality filter at all.
    """
    # Use hand_strength_2 as fast approximation (launch tradeoff)
    # This is the "two-tier optimization" from the spec:
    # Fast pre-filter, accepts minor inaccuracy
    return hand_strength_2_lookup(keep_2)


# ── Optimal discard ──────────────────────────────────────────────────────────

def optimal_discard(hole_5, flop_3, opp_weights_fn=None):
    """
    Evaluate all C(5,2)=10 keep options with exact equity.
    Returns (index_i, index_j) of the two cards to KEEP.

    Includes discard stochasticity (Defense 1): if top-2 keeps are within
    0.03 equity, takes #2 with 20% probability to hide information.
    """
    keeps = []
    for i, j in combinations(range(5), 2):
        keep = (hole_5[i], hole_5[j])
        discard = [hole_5[k] for k in range(5) if k not in (i, j)]
        dead = tuple(discard)
        eq = exact_equity(keep, flop_3, dead, opp_weights_fn=opp_weights_fn)
        keeps.append((eq, i, j))

    keeps.sort(key=lambda x: x[0], reverse=True)
    best_eq, bi, bj = keeps[0]
    second_eq, si, sj = keeps[1]

    # Defense 1: Discard stochasticity
    if best_eq - second_eq < 0.03 and random.random() < 0.20:
        return (si, sj)
    return (bi, bj)
```

```python
# submission/opponent.py
"""
Opponent model: observes all opponent actions, exposes clean context dict.
Never makes decisions — only observes and reports.
"""
import math
from collections import deque
from submission.card_utils import rank, suit, is_suited, is_paired, is_connected, is_high, classify_hand


class OpponentModel:
    def __init__(self):
        # ── Per-street action counts ─────────────────────────────────────
        # street 0=preflop, 1=flop, 2=turn, 3=river
        self._raises = [0, 0, 0, 0]
        self._calls = [0, 0, 0, 0]
        self._checks = [0, 0, 0, 0]
        self._folds = [0, 0, 0, 0]
        self._actions_total = [0, 0, 0, 0]

        # ── Fold response to OUR raises ──────────────────────────────────
        self._folds_to_our_raise = [0, 0, 0, 0]
        self._calls_to_our_raise = [0, 0, 0, 0]

        # ── Bet sizing history ───────────────────────────────────────────
        self._raise_sizes = deque(maxlen=200)  # (raise_amount, pot_size)

        # ── Bluff rate (Bayesian Beta distribution) ──────────────────────
        self._bluff_alpha = 2.0   # observed bluffs
        self._bluff_beta = 8.0    # observed value hands

        # ── Per-hand action sequence ─────────────────────────────────────
        self._hand_actions = []      # e.g., ['CHECK', 'RAISE', 'RAISE']
        self._raised_this_hand = False

        # ── Range preference weights (learned from discards + showdowns) ─
        self.pref_suited = 1.0
        self.pref_paired = 1.0
        self.pref_connected = 1.0
        self.pref_high = 1.0

        # ── Showdown hand type tracking ──────────────────────────────────
        self._sd_hand_types = {}    # {'flush': 5, 'pair': 8, ...}
        self._sd_count = 0

        # ── Discard pattern tracking ─────────────────────────────────────
        self._discard_data = {
            'kept_suited_count': 0,
            'kept_pair_count': 0,
            'discarded_pair_count': 0,
            'total_observed': 0,
        }

    # ── Action updates ───────────────────────────────────────────────────

    def update_action(self, action: str, street: int, bet_size: int = 0, pot: int = 0):
        """Record an opponent action."""
        s = min(street, 3)
        self._actions_total[s] += 1

        if 'RAISE' in action:
            self._raises[s] += 1
            self._raised_this_hand = True
            if pot > 0 and bet_size > 0:
                self._raise_sizes.append((bet_size, pot))
            self._hand_actions.append('RAISE')
        elif 'CALL' in action:
            self._calls[s] += 1
            self._hand_actions.append('CALL')
        elif 'CHECK' in action:
            self._checks[s] += 1
            self._hand_actions.append('CHECK')
        elif 'FOLD' in action:
            self._folds[s] += 1
            self._hand_actions.append('FOLD')

    def update_fold_to_our_raise(self, street: int):
        s = min(street, 3)
        self._folds_to_our_raise[s] += 1

    def update_call_to_our_raise(self, street: int):
        s = min(street, 3)
        self._calls_to_our_raise[s] += 1

    # ── Showdown updates ─────────────────────────────────────────────────

    def update_showdown(self, raised: bool = False, hand_strong: bool = False,
                        hand_score: int = None, raised_streets: list = None):
        """Update bluff rate and hand type tracking from showdown."""
        if raised:
            if hand_strong:
                self._bluff_beta += 1.0
            else:
                self._bluff_alpha += 1.0

        if hand_score is not None:
            hand_type = classify_hand(hand_score)
            self._sd_hand_types[hand_type] = self._sd_hand_types.get(hand_type, 0) + 1
            self._sd_count += 1

    # ── Discard analysis ─────────────────────────────────────────────────

    def update_discards(self, opp_discards):
        """Analyze opponent's 3 discarded cards to infer keep preferences."""
        if len(opp_discards) != 3 or any(c == -1 for c in opp_discards):
            return

        self._discard_data['total_observed'] += 1
        ranks = [rank(c) for c in opp_discards]
        suits = [suit(c) for c in opp_discards]

        from collections import Counter
        rank_counts = Counter(ranks)
        suit_counts = Counter(suits)

        # Discarded a pair → kept something better
        if max(rank_counts.values()) >= 2:
            self._discard_data['discarded_pair_count'] += 1

        # All 3 different suits discarded → kept cards likely suited
        if len(suit_counts) == 3:
            self._discard_data['kept_suited_count'] += 1
            self.pref_suited = min(2.0, self.pref_suited + 0.02)

        # No pairs in discard + not all different suits → likely kept pair
        if max(rank_counts.values()) == 1 and len(suit_counts) < 3:
            self._discard_data['kept_pair_count'] += 1
            self.pref_paired = min(2.0, self.pref_paired + 0.02)

    # ── Hand reset ───────────────────────────────────────────────────────

    def reset_hand(self):
        """Reset per-hand state. Call at end of each hand."""
        self._hand_actions = []
        self._raised_this_hand = False

    # ── Query methods ────────────────────────────────────────────────────

    def _total_postflop_actions(self) -> int:
        return sum(self._actions_total[1:4])

    def _total_postflop_raises(self) -> int:
        return sum(self._raises[1:4])

    def get_context(self, street: int) -> dict:
        """Return all opponent intelligence for strategy module."""
        total_actions = sum(self._actions_total)
        total_raises = sum(self._raises)
        pf_actions = self._actions_total[0]
        pf_raises = self._raises[0]

        raise_rate = total_raises / max(1, total_actions)
        pf_raise_rate = pf_raises / max(1, pf_actions)

        # Post-flop fold rate to our raises
        total_our_raises_faced = sum(self._folds_to_our_raise[s] + self._calls_to_our_raise[s] for s in range(4))
        total_folds_to_us = sum(self._folds_to_our_raise)
        fold_rate = total_folds_to_us / max(1, total_our_raises_faced)

        # Bluff rate (Bayesian)
        bluff_rate = self._bluff_alpha / (self._bluff_alpha + self._bluff_beta)

        # Average raise sizing
        avg_raise_frac = self.avg_raise_frac()

        # Opponent type
        raises_often = raise_rate > 0.40
        folds_often = fold_rate > 0.40

        if raises_often and not folds_often:
            opp_type = 'lag'
        elif raises_often and folds_often:
            opp_type = 'tag'
        elif not raises_often and not folds_often:
            opp_type = 'station'
        else:
            opp_type = 'nit'

        return {
            'raise_rate': raise_rate,
            'pf_raise_rate': pf_raise_rate,
            'raises_often': raises_often,
            'fold_rate_to_our_raise': fold_rate,
            'folds_often': folds_often,
            'pf_fold_rate': self._folds[0] / max(1, pf_actions),
            'bluff_rate': bluff_rate,
            'avg_raise_frac': avg_raise_frac,
            'is_trap_line': self.is_trap_line(),
            'is_barrel_line': self.is_barrel_line(),
            'opponent_type': opp_type,
            'raised_this_hand': self._raised_this_hand,
            'total_actions': total_actions,
            'sd_count': self._sd_count,
        }

    def avg_raise_frac(self) -> float:
        if len(self._raise_sizes) < 5:
            return 0.50
        return sum(sz / max(1, p) for sz, p in self._raise_sizes) / len(self._raise_sizes)

    def is_trap_line(self) -> bool:
        """Detect check-then-raise pattern (slowplay/trap)."""
        if len(self._hand_actions) < 2:
            return False
        return 'CHECK' in self._hand_actions[:-1] and self._hand_actions[-1] == 'RAISE'

    def is_barrel_line(self) -> bool:
        """Detect multi-barrel: raised on 2+ streets."""
        return sum(1 for a in self._hand_actions if a == 'RAISE') >= 2

    def opp_weights_fn(self, hand: tuple) -> float:
        """Weight for a possible opponent 2-card holding based on learned preferences."""
        c1, c2 = hand
        w = 1.0
        if is_suited(c1, c2):
            w *= self.pref_suited
        if is_paired(c1, c2):
            w *= self.pref_paired
        if is_connected(c1, c2):
            w *= self.pref_connected
        if is_high(c1) or is_high(c2):
            w *= self.pref_high
        return w
```

```python
# submission/match_manager.py
"""
Match Manager: tracks match state and provides continuous phase signals.
All transitions are smooth and jittered — no hard phase boundaries.
"""
import math
import random


class MatchManager:
    TOTAL_HANDS = 1000
    BLEED_RATE = 1.5  # avg chips lost per hand from blinds

    def __init__(self):
        self._hand_number = 0
        self._cumulative_reward = 0
        self._time_left = 500.0  # conservative default (Phase 1)
        self._time_used = 0.0

        # Per-match persona rotation (Defense 2)
        self.persona = {
            'value_raise_thr': random.uniform(0.58, 0.66),
            'bluff_cap': random.uniform(0.25, 0.50),
            'pf_call_margin': random.uniform(0.03, 0.07),
            'probe_thr_base': random.uniform(0.40, 0.50),
            'call_thr_margin': random.uniform(0.02, 0.05),
            'bluff_size_center': random.uniform(0.55, 0.65),
            'value_size_center': random.uniform(0.70, 0.80),
        }

    def update(self, reward: float, time_used: float = 0.0, time_left: float = 0.0):
        """Call at end of each hand."""
        self._hand_number += 1
        self._cumulative_reward += reward
        if time_left > 0:
            self._time_left = time_left
        if time_used > 0:
            self._time_used = time_used

    def get_state(self) -> dict:
        remaining = max(1, self.TOTAL_HANDS - self._hand_number)
        remaining_bleed = remaining * self.BLEED_RATE

        # Pressure: negative = comfortable lead, positive = desperate
        if remaining_bleed > 0:
            pressure = -self._cumulative_reward / remaining_bleed
            pressure = max(-1.0, min(1.0, pressure))
        else:
            pressure = 0.0

        # Aggression: high early, decays smoothly (with jitter)
        early_factor = max(0.0, 1.0 - self._hand_number / 300.0)
        early_factor *= (1.0 + random.uniform(-0.20, 0.20))
        early_factor = max(0.0, early_factor)

        # Combine with desperation: if behind, stay aggressive
        aggression = max(early_factor, max(0.0, pressure) * 0.8)
        aggression = min(1.0, aggression)

        # Protection mode: lead > 15% of remaining bleed
        in_protection = (self._cumulative_reward > remaining_bleed * 0.15
                         and self._hand_number > 200)

        # Lockout: lead > remaining_hands * 1.5 + buffer
        lockout_threshold = remaining * 1.5 + 30
        in_lockout = (self._cumulative_reward > lockout_threshold
                      and self._hand_number > 400)

        # Suppress aggression in protection mode
        if in_protection:
            aggression *= 0.3

        return {
            'hand_number': self._hand_number,
            'cumulative_reward': self._cumulative_reward,
            'remaining_hands': remaining,
            'pressure': pressure,
            'aggression': aggression,
            'in_protection': in_protection,
            'in_lockout': in_lockout,
            'time_left': self._time_left,
            'time_per_hand': self._time_left / max(1, remaining),
        }
```

```python
# submission/strategy.py
"""
Adaptive strategy engine.
Uses sigmoid probabilistic thresholds, GTO-anchored bluff frequency,
board-texture sizing, and range balance enforcement.
"""
import math
import random
from collections import Counter
from submission.card_utils import suit, rank


# ── Sigmoid threshold ────────────────────────────────────────────────────────

def soft_decision(value: float, threshold: float, temp: float = 0.04) -> float:
    """P(action) = sigmoid((value - threshold) / temp)."""
    x = (value - threshold) / temp
    x = max(-10.0, min(10.0, x))
    return 1.0 / (1.0 + math.exp(-x))


# ── Board-texture sizing ────────────────────────────────────────────────────

def _ace_wrap_connected(r1: int, r2: int) -> bool:
    """Check if two ranks are connected, handling Ace wraps in 27-card deck."""
    diff = abs(r1 - r2)
    if diff <= 2:
        return True
    # Ace (rank 8) wraps: A-2, A-3 connected (low); A-7, A-8, A-9 connected (high, Ace as 10)
    if r1 == 8 or r2 == 8:
        other = r2 if r1 == 8 else r1
        return other <= 1 or other >= 6
    return False


def bet_size(pot: int, board: list, persona: dict) -> int:
    """Size based on board wetness, not hand strength (Defense 4)."""
    if not board or pot <= 0:
        return max(2, pot // 2)

    suits_list = [suit(c) for c in board if c >= 0]
    ranks_list = sorted([rank(c) for c in board if c >= 0])

    if not suits_list:
        return max(2, pot // 2)

    # Board wetness
    flush_score = max(Counter(suits_list).values()) / len(suits_list)
    connect_score = 0
    if len(ranks_list) > 1:
        # CRITICAL: Use ace-wrap-aware connectivity
        # Without this, A-2-3 board (ranks [0,1,8]) wrongly scores 0 connectivity
        connect_score = sum(
            1 for i in range(len(ranks_list) - 1)
            if _ace_wrap_connected(ranks_list[i], ranks_list[i + 1])
        ) / (len(ranks_list) - 1)
    wetness = (flush_score + connect_score) / 2

    # Base sizing: wet boards → larger, dry boards → smaller
    base = persona.get('bluff_size_center', 0.60) + wetness * 0.20
    base = max(0.40, min(0.95, base))

    # Jitter ±20%
    result = int(pot * base * (1.0 + random.uniform(-0.20, 0.20)))
    return max(2, result)


# ── Bluff EV math ────────────────────────────────────────────────────────────

def compute_bluff_ev(fold_rate: float, pot: int, raise_cost: int, equity: float) -> float:
    """Full bluff EV including showdown equity when called."""
    total_pot_if_called = pot + 2 * raise_cost
    return (fold_rate * pot) + ((1 - fold_rate) * (equity * total_pot_if_called - raise_cost))


def gto_bluff_ratio(bet_frac: float) -> float:
    """GTO bluff ceiling: bettor's optimal bluff frequency for indifference."""
    return bet_frac / (1.0 + 2 * bet_frac)


# ── Main decision function ───────────────────────────────────────────────────

def decide(equity: float, pot_odds: float, street: int, pot: int,
           cost: int, min_raise: int, max_raise: int,
           valid_actions: list, opp_ctx: dict, match_state: dict,
           board: list, persona: dict, facing_raise: bool) -> tuple:
    """
    Returns action tuple: (action_type, raise_amount, 0, 0).
    action_type: 0=FOLD, 1=RAISE, 2=CHECK, 3=CALL
    """

    # ── Lockout: auto-fold everything ────────────────────────────────────
    if match_state.get('in_lockout', False):
        if valid_actions[2]:  # can check
            return (2, 0, 0, 0)
        return (0, 0, 0, 0)

    # ── Protection mode: only play strong hands ──────────────────────────
    if match_state.get('in_protection', False) and equity < 0.60:
        if valid_actions[2]:
            return (2, 0, 0, 0)
        return (0, 0, 0, 0)

    aggression = match_state.get('aggression', 0.0)
    fold_rate = opp_ctx.get('fold_rate_to_our_raise', 0.3)
    bluff_rate = opp_ctx.get('bluff_rate', 0.2)
    folds_often = opp_ctx.get('folds_often', False)
    raises_often = opp_ctx.get('raises_often', False)

    # ── Pre-flop decisions (street 0) ────────────────────────────────────
    if street == 0:
        return _decide_preflop(equity, pot_odds, valid_actions, cost,
                               min_raise, max_raise, pot, opp_ctx,
                               match_state, persona)

    # ── Post-flop decisions ──────────────────────────────────────────────

    # Layer A: Should I raise? (Aggression)
    if valid_actions[1] and max_raise >= min_raise:
        raise_action = _layer_a_raise(equity, pot, board, persona, fold_rate,
                                       opp_ctx, match_state, min_raise, max_raise,
                                       facing_raise)
        if raise_action is not None:
            return raise_action

    # Layer B: Should I call their raise? (Defense)
    if facing_raise and valid_actions[3]:
        call_action = _layer_b_call(equity, pot_odds, cost, pot, bluff_rate,
                                     opp_ctx, match_state, persona)
        if call_action is not None:
            return call_action

    # Layer C: Should I probe? (Initiative)
    if not facing_raise and valid_actions[1] and max_raise >= min_raise:
        probe_action = _layer_c_probe(equity, pot, board, persona, fold_rate,
                                       opp_ctx, match_state, min_raise, max_raise)
        if probe_action is not None:
            return probe_action

    # Default: check if possible, else fold
    if valid_actions[2]:
        return (2, 0, 0, 0)
    if facing_raise and valid_actions[0]:
        return (0, 0, 0, 0)
    return (2, 0, 0, 0) if valid_actions[2] else (0, 0, 0, 0)


# ── Pre-flop ─────────────────────────────────────────────────────────────────

def _decide_preflop(equity, pot_odds, valid_actions, cost, min_raise, max_raise,
                    pot, opp_ctx, match_state, persona):
    aggression = match_state.get('aggression', 0.0)
    pf_margin = persona.get('pf_call_margin', 0.05)

    # Call threshold: pot_odds + margin, adjusted by aggression and opp raise rate
    call_thr = pot_odds + pf_margin
    call_thr -= aggression * 0.06  # more aggressive early
    call_thr -= min(0.10, max(0.0, opp_ctx.get('pf_raise_rate', 0.3) - 0.40) * 0.30)
    call_thr = max(0.15, call_thr)  # absolute floor

    if cost > 0 and valid_actions[3]:
        # Facing a raise
        p_call = soft_decision(equity, call_thr)
        if random.random() < p_call:
            return (3, 0, 0, 0)  # CALL
        return (0, 0, 0, 0)  # FOLD

    # Not facing a raise
    if valid_actions[2]:
        # Can check — consider raising with strong hands
        raise_thr = persona.get('value_raise_thr', 0.62) - aggression * 0.04
        p_raise = soft_decision(equity, raise_thr)
        if random.random() < p_raise and valid_actions[1] and max_raise >= min_raise:
            raise_amt = max(min_raise, min(max_raise, int(pot * 0.75)))
            return (1, raise_amt, 0, 0)
        return (2, 0, 0, 0)  # CHECK

    # Can only call or fold
    if valid_actions[3]:
        p_call = soft_decision(equity, call_thr)
        if random.random() < p_call:
            return (3, 0, 0, 0)
    return (0, 0, 0, 0)


# ── Layer A: Raise (Aggression) ──────────────────────────────────────────────

def _layer_a_raise(equity, pot, board, persona, fold_rate, opp_ctx,
                   match_state, min_raise, max_raise, facing_raise):
    """Determine if we should raise (value or bluff)."""
    aggression = match_state.get('aggression', 0.0)
    value_thr = persona.get('value_raise_thr', 0.62) - aggression * 0.04

    # If facing a re-raise, require higher equity (raise war cap)
    if facing_raise:
        reraise_thr = 0.65 - min(0.10, opp_ctx.get('bluff_rate', 0.2) * 0.5)
        value_thr = max(value_thr, reraise_thr)

    # Value raise
    p_value = soft_decision(equity, value_thr)
    if random.random() < p_value:
        raise_amt = bet_size(pot, board, persona)
        raise_amt = max(min_raise, min(max_raise, raise_amt))
        return (1, raise_amt, 0, 0)

    # Bluff raise (only if not facing a raise, and has some equity)
    if not facing_raise:
        street = len([c for c in board if c >= 0])
        min_bluff_eq = 0.10 if street <= 4 else 0.20  # flop/turn vs river
        if equity >= min_bluff_eq:
            raise_amt = bet_size(pot, board, persona)
            raise_amt = max(min_raise, min(max_raise, raise_amt))
            raise_cost = raise_amt
            bet_frac = raise_cost / max(1, pot)

            bluff_ev = compute_bluff_ev(fold_rate, pot, raise_cost, equity)

            if bluff_ev > 0:
                # GTO-anchored ceiling
                gto_ceiling = gto_bluff_ratio(bet_frac)
                exploit_prob = min(1.0, bluff_ev / max(1, pot)) * persona.get('bluff_cap', 0.35)

                # Confidence blend
                total_obs = opp_ctx.get('total_actions', 0)
                confidence = min(1.0, total_obs / 30.0)
                bluff_prob = exploit_prob * confidence + gto_ceiling * (1 - confidence)
                bluff_prob = min(bluff_prob, gto_ceiling * 1.3)

                if random.random() < bluff_prob:
                    return (1, raise_amt, 0, 0)

    return None


# ── Layer B: Call (Defense) ──────────────────────────────────────────────────

def _layer_b_call(equity, pot_odds, cost, pot, bluff_rate, opp_ctx,
                  match_state, persona):
    """Determine if we should call a raise."""
    aggression = match_state.get('aggression', 0.0)
    margin = persona.get('call_thr_margin', 0.03)

    # Core formula: bluff-rate-adjusted call threshold
    call_thr = (pot_odds + margin) * (1.0 - bluff_rate)

    # Modifiers
    call_thr -= aggression * 0.04  # call wider early

    # Trap line → respect more
    if opp_ctx.get('is_trap_line', False):
        call_thr += 0.05

    # Barrel line from hyper-raiser → less respect
    if opp_ctx.get('is_barrel_line', False) and opp_ctx.get('raises_often', False):
        call_thr -= 0.04

    # Sizing tell
    avg_frac = opp_ctx.get('avg_raise_frac', 0.50)
    current_frac = cost / max(1, pot - cost)
    if current_frac < 0.30 and avg_frac < 0.40:
        call_thr -= 0.04  # min-raiser probing
    elif current_frac > 0.80:
        call_thr += 0.03  # big raise = value

    # Floor: never fold getting >5:1 with any equity
    call_thr = max(0.10, call_thr)

    p_call = soft_decision(equity, call_thr)
    if random.random() < p_call:
        return (3, 0, 0, 0)

    return None  # fold via default


# ── Layer C: Probe (Initiative) ──────────────────────────────────────────────

def _layer_c_probe(equity, pot, board, persona, fold_rate, opp_ctx,
                   match_state, min_raise, max_raise):
    """Determine if we should probe (bet when checked to us)."""
    probe_thr = persona.get('probe_thr_base', 0.45)
    aggression = match_state.get('aggression', 0.0)

    # vs folders: probe much wider
    if fold_rate > 0.60:
        probe_thr = max(0.20, probe_thr - (fold_rate - 0.40) * 0.50)

    probe_thr -= aggression * 0.06

    p_probe = soft_decision(equity, probe_thr)
    if random.random() < p_probe:
        raise_amt = bet_size(pot, board, persona)
        raise_amt = max(min_raise, min(max_raise, raise_amt))
        return (1, raise_amt, 0, 0)

    return None
```

```python
# submission/player.py
"""
Poker Bot v2 — PlayerAgent orchestrator.
Thin wiring layer connecting equity, opponent, strategy, and match_manager modules.
"""
import os
import sys
import random

# Ensure submission package is importable
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from agents.agent import Agent
from submission.equity import exact_equity, optimal_discard, preflop_strength, hand_rank
from submission.opponent import OpponentModel
from submission.strategy import decide
from submission.match_manager import MatchManager
from submission.card_utils import classify_hand


class PlayerAgent(Agent):
    def __name__(self):
        return "PlayerAgent"

    def __init__(self, stream: bool = False, player_id: str = None):
        super().__init__(stream=stream, player_id=player_id)
        self.opponent = OpponentModel()
        self.match_mgr = MatchManager()
        # Cache evaluator for showdown processing (avoid creating PokerEnv per hand)
        from gym_env import PokerEnv
        self._evaluator = PokerEnv().evaluator

        # Per-hand state
        self._my_cards = []
        self._my_kept = None       # (card1, card2) after discard
        self._my_discards = None   # (card1, card2, card3) after discard
        self._board = []
        self._opp_discards = None
        self._street = -1
        self._last_street = -1
        self._we_raised = False
        self._we_raised_this_hand = False
        self._hands_played = 0

        # Showdown poison tracking (Defense 3)
        self._poison_active = False

    def act(self, observation, reward, terminated, truncated, info) -> tuple:
        obs = observation

        # ── Terminal call: no-op (observe() handles end-of-hand updates) ─
        # NOTE: sim.py calls BOTH observe() and act() with terminated=True.
        # All end-of-hand logic (match manager update, opponent reset, hand
        # state reset) lives in observe() to avoid double-counting.
        if terminated:
            return (0, 0, 0, 0)  # dummy action, hand is over

        # ── Parse observation ────────────────────────────────────────────
        street = obs['street']
        valid = obs['valid_actions']
        my_cards = [c for c in obs['my_cards'] if c != -1]
        community = [c for c in obs['community_cards'] if c != -1]
        opp_discards = [c for c in obs['opp_discarded_cards'] if c != -1]
        my_discards_obs = [c for c in obs['my_discarded_cards'] if c != -1]
        my_bet = obs['my_bet']
        opp_bet = obs['opp_bet']
        min_raise = obs['min_raise']
        max_raise = obs['max_raise']
        opp_last = obs.get('opp_last_action', 'None')

        # Track opponent actions
        if opp_last and opp_last != 'None' and opp_last != 'DISCARD':
            opp_raise_size = max(0, opp_bet - my_bet) if 'RAISE' in opp_last else 0
            pot = my_bet + opp_bet
            self.opponent.update_action(opp_last, street, opp_raise_size, pot)

            # Track fold/call response to our raises
            if self._we_raised:
                if 'FOLD' in opp_last:
                    self.opponent.update_fold_to_our_raise(street)
                elif 'CALL' in opp_last:
                    self.opponent.update_call_to_our_raise(street)
                self._we_raised = False

        # Update board and cards
        self._my_cards = my_cards
        self._board = community
        self._street = street

        # Track opponent discards
        if len(opp_discards) == 3 and self._opp_discards is None:
            self._opp_discards = tuple(opp_discards)
            self.opponent.update_discards(opp_discards)

        # Track our discards
        if len(my_discards_obs) == 3 and self._my_discards is None:
            self._my_discards = tuple(my_discards_obs)

        # ── Lockout check ────────────────────────────────────────────────
        match_state = self.match_mgr.get_state()
        if match_state['in_lockout']:
            if valid[2]:  # check
                return (2, 0, 0, 0)
            return (0, 0, 0, 0)  # fold

        # ── DISCARD action (street 1, first action) ──────────────────────
        if valid[4]:  # DISCARD required
            hole_5 = tuple(my_cards[:5])
            flop_3 = tuple(community[:3])
            weights_fn = self.opponent.opp_weights_fn
            keep_i, keep_j = optimal_discard(hole_5, flop_3, weights_fn)
            self._my_kept = (hole_5[keep_i], hole_5[keep_j])
            return (4, 0, keep_i, keep_j)

        # ── BETTING action ───────────────────────────────────────────────
        # Determine our 2 kept cards
        if self._my_kept is not None:
            my_2 = self._my_kept
        elif len(my_cards) == 2:
            my_2 = tuple(my_cards)
        else:
            my_2 = tuple(my_cards[:2])

        pot = my_bet + opp_bet
        cost = opp_bet - my_bet  # cost to call
        pot_odds = cost / max(1, pot + cost)
        facing_raise = cost > 0

        # Compute equity
        dead_cards = list(self._my_discards or []) + list(self._opp_discards or [])
        if street == 0:
            # Pre-flop: use preflop_strength
            eq = preflop_strength(tuple(my_cards[:5])) if len(my_cards) == 5 else 0.5
        else:
            eq = exact_equity(
                my_2, tuple(community), tuple(dead_cards),
                opp_discards=self._opp_discards,
                my_discards=self._my_discards,
                opp_weights_fn=self.opponent.opp_weights_fn,
            )

        # Get opponent context
        opp_ctx = self.opponent.get_context(street)

        # ── Showdown poison (Defense 3) ──────────────────────────────────
        import math

        # If poison already active from earlier street, check-call small bets
        if self._poison_active:
            if facing_raise:
                raise_cost = opp_bet - my_bet
                # ABORT CIRCUIT: if opponent bombs pot, poison is too expensive
                if raise_cost > pot * 0.40 or raise_cost > 15:
                    self._poison_active = False  # revert to standard strategy
                    # Fall through to normal decision logic below
                elif valid[3]:
                    return (3, 0, 0, 0)  # call small bet — force showdown
                else:
                    self._poison_active = False  # can't call, abort
            else:
                if valid[2]:
                    return (2, 0, 0, 0)  # CHECK
                elif valid[3]:
                    return (3, 0, 0, 0)  # CALL
                return (0, 0, 0, 0)

        # Vector A: poison in passive pots
        if street >= 2 and not facing_raise:
            poison_prob = 0.03 * math.exp(-pot / 15.0)
            if random.random() < poison_prob and eq < 0.35:
                self._poison_active = True
                if valid[2]:
                    return (2, 0, 0, 0)
                elif valid[3]:
                    return (3, 0, 0, 0)

        # Vector B: poison in raised pots where WE were aggressor
        if street >= 2 and self._we_raised_this_hand and not facing_raise:
            poison_prob = 0.015 * math.exp(-pot / 20.0)
            if random.random() < poison_prob and eq < 0.30:
                self._poison_active = True  # Flag forces check-call through showdown
                if valid[2]:
                    return (2, 0, 0, 0)

        # ── Strategy decision ────────────────────────────────────────────
        action = decide(
            equity=eq,
            pot_odds=pot_odds,
            street=street,
            pot=pot,
            cost=cost,
            min_raise=min_raise,
            max_raise=max_raise,
            valid_actions=valid,
            opp_ctx=opp_ctx,
            match_state=match_state,
            board=community,
            persona=self.match_mgr.persona,
            facing_raise=facing_raise,
        )

        # Track if we raised
        if action[0] == 1:  # RAISE
            self._we_raised = True
            self._we_raised_this_hand = True

        return action

    def observe(self, observation, reward, terminated, truncated, info):
        """Called after opponent's action when it's not our turn, and at hand end."""
        if terminated:
            # Update match manager
            self.match_mgr.update(
                reward=reward,
                time_used=observation.get('time_used', 0),
                time_left=observation.get('time_left', 0),
            )

            # Process showdown if available
            if 'player_0_cards' in info:
                self._process_showdown(info, observation)

            self.opponent.reset_hand()
            self._reset_hand_state()
            self._hands_played += 1

    def _process_showdown(self, info, obs):
        """Extract opponent hand info from showdown and update model."""
        from gym_env import PokerEnv
        try:
            # Determine which player we are
            # player_0_cards and player_1_cards are string lists like ['Ad', '9h']
            community = [c for c in obs['community_cards'] if c != -1]
            if len(community) != 5:
                return

            # Get opponent's cards as strings
            my_cards = [c for c in obs['my_cards'] if c != -1]

            # Determine if we're player 0 or player 1
            p0_cards = info.get('player_0_cards', [])
            p1_cards = info.get('player_1_cards', [])

            # Convert our cards to strings for comparison
            my_strs = set(PokerEnv.int_card_to_str(c) for c in my_cards)

            if set(p0_cards) == my_strs:
                opp_strs = p1_cards
            elif set(p1_cards) == my_strs:
                opp_strs = p0_cards
            else:
                return  # can't determine

            if len(opp_strs) != 2:
                return

            # Evaluate opponent hand using cached evaluator
            from treys import Card
            opp_treys = [Card.new(s) for s in opp_strs]
            board_treys = [PokerEnv.int_to_card(c) for c in community]
            opp_score = self._evaluator.evaluate(opp_treys, board_treys)

            # Determine if opponent's hand is strong
            hand_type = classify_hand(opp_score)
            is_strong = hand_type in ('straight_flush', 'full_house', 'flush', 'straight', 'trips')

            self.opponent.update_showdown(
                raised=self.opponent._raised_this_hand,
                hand_strong=is_strong,
                hand_score=opp_score,
            )
        except Exception:
            pass  # silently handle any parsing errors

    def _reset_hand_state(self):
        self._my_cards = []
        self._my_kept = None
        self._my_discards = None
        self._board = []
        self._opp_discards = None
        self._street = -1
        self._last_street = -1
        self._we_raised = False
        self._we_raised_this_hand = False
        self._poison_active = False


if __name__ == "__main__":
    PlayerAgent.run()
```

This table provides exact pre-flop 5-card strength using flop-aware evaluation. It's computationally expensive (~hours) and is NOT required for v2 launch — the fallback uses `hand_strength_2` for pre-flop evaluation.

**When to do this:** After v2 is submitted and confirmed working in tournament.

### Task 12 (future): Add `strategy_params.json` and `opponent_priors.json`

Externalizes all strategy thresholds into tunable config files. Enables rapid iteration without code changes.

**When to do this:** After analyzing first round of v2 tournament results.

### Task 13 (future): Vectorize equity engine with NumPy broadcasting

Replace Python-loop equity enumeration with vectorized NumPy operations for 10x speedup. This also unblocks the flop-aware rationality filter (currently using hand_strength_2 approximation as a launch tradeoff — see `_quick_flop_equity()` comment in equity.py).

**When to do this:** If timing becomes a bottleneck in Phase 2/3, or to fix the rationality filter approximation.

### Task 14 (future): Add `analyze_logs.py` for offline tournament analysis

Processes tournament game logs to produce: threshold calibration (identify folds where equity > call threshold), bluff efficiency tracking, opponent scouting (build/update opponent_priors.json), discard quality audit, and strategy regression detection.

**When to do this:** After first round of v2 tournament results.

### Task 15 (future): Add range balance enforcement per board-texture class

Track bluff-to-value ratio per board type (wet/medium/dry) and suppress bluffs when ratio exceeds GTO by >15%. Prevents opponents from profiling range composition per board type.

**When to do this:** After confirming basic strategy is working in tournament.

## Intentional Deferrals (spec features not in v2 launch)

These spec features are intentionally deferred to post-launch iterations:
- **Flop-aware rationality filter** — uses hand_strength_2 approximation (Task 13 unblocks this)
- **hand_potential_5.npy** — falls back to max-of-10-keeps from hand_strength_2 (Task 11)
- **strategy_params.json** — thresholds hardcoded in strategy.py (Task 12)
- **opponent_priors.json** — starts from neutral priors each match (Task 12)
- **Range balance enforcement** — bluff frequency capped by GTO ceiling but not tracked per board type (Task 15)
- **analyze_logs.py** — manual log analysis for now (Task 14)
- **NumPy vectorized equity** — uses Python loops, adequate for Phase 1 timing (Task 13)
